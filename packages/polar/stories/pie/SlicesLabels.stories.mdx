import { ArgsTable, Canvas, Meta, Story } from '@storybook/addon-docs'
import { SlicesLabels } from '../../src'
import { ChartPieSlicesDecorator, ChartDoughnutSlicesDecorator } from './decorators'

# SlicesLabels

<Meta title="Addons/Polar/Pie/SlicesLabels" />

export const Template = args => <SlicesLabels {...args} />

`SlicesLabels` draws pieces for a pie / doughnut chart.

<ArgsTable of={SlicesLabels} />

## Radial position

Prop `r` determines the radial position of the labels.

<Canvas>
    <Story
        name="radial position, pie"
        args={{ r: 0.8 }}
        component={SlicesLabels}
        decorators={[ChartPieSlicesDecorator]}
    >
        {Template.bind({})}
    </Story>
    <Story
        name="radial position, doughnut"
        args={{ r: 0.8 }}
        component={SlicesLabels}
        decorators={[ChartDoughnutSlicesDecorator]}
    >
        {Template.bind({})}
    </Story>
</Canvas>

## Data subsets

By default, `SlicesLabels` draws labels for all slices in a dataset. To draw subsets, set props `ids`.

<Canvas>
    <Story
        name="selected ids"
        args={{ ids: ['alpha', 'gamma'], r: 0.6 }}
        component={SlicesLabels}
        decorators={[ChartPieSlicesDecorator]}
    >
        {Template.bind({})}
    </Story>
</Canvas>

## Minimum angle

Prop `minAngle` specifies an angle threshold. Labels for slices smaller than the threshold will be omitted.

<Canvas>
    <Story
        name="minimum angle"
        args={{ minAngle: 45, r: 0.6 }}
        component={SlicesLabels}
        decorators={[ChartPieSlicesDecorator]}
    >
        {Template.bind({})}
    </Story>
</Canvas>

## Styling

Labels can be styled using css.

<Canvas>
    <Story
        name="styled"
        args={{ r: 0.6, style: { fill: '#ffffff', fontSize: '14px', fontWeight: 600 } }}
        component={SlicesLabels}
        decorators={[ChartPieSlicesDecorator]}
    >
        {Template.bind({})}
    </Story>
</Canvas>

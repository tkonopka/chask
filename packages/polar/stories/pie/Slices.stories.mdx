import { ArgsTable, Canvas, Meta, Story } from '@storybook/addon-docs'
import { Slices } from '../../src'
import { ChartPieDecorator, ChartDoughnutDecorator } from './decorators'

# Slices

<Meta title="Addons/Polar/Pie/Slices" />

export const Template = args => <Slices {...args} />

`Slices` draws pieces for a pie / doughnut chart.

<ArgsTable of={Slices} />

## Data subsets

By default, `Slices` draws an entire dataset. To draw subsets, set props `ids`.

<Canvas>
    <Story
        name="selected ids"
        args={{ ids: ['alpha', 'gamma'] }}
        component={Slices}
        decorators={[ChartDoughnutDecorator]}
    >
        {Template.bind({})}
    </Story>
</Canvas>

## Decorations

Prop `rCorner` creates slices with rounded corners.
Prop `padAngle` creates empty space between adjacent slices.

<Canvas>
    <Story
        name="rounded corners"
        args={{
            r: 6,
            padAngle: 0.05,
        }}
        component={Slices}
        decorators={[ChartDoughnutDecorator]}
    >
        {Template.bind({})}
    </Story>
</Canvas>

## Styling

Slices can be styled using css.

<Canvas>
    <Story
        name="styled"
        args={{ style: { strokeWidth: 2, stroke: '#222', fillOpacity: 0.7 } }}
        component={Slices}
        decorators={[ChartPieDecorator]}
    >
        {Template.bind({})}
    </Story>
</Canvas>
